Интерактивные задания

Написать и выполнить C++ код

Пример кода C++:

console.log("Привет, мир!");

let x = 5;
let y = 10;
console.log("Сумма чисел " + x + " и " + y + " равна: " + (x + y));

// Пример цикла
for(let i = 0; i < 5; i++) {
    console.log("Итерация: " + i);
}

// Пример массива
let arr = [1, 2, 3, 4, 5];
console.log("Элементы массива:");
for(let i = 0; i < arr.length; i++) {
    console.log(arr[i]);
}

Написать и выполнить C# код

Пример кода C#:

console.log("Привет, мир!");

let x = 5;
let y = 10;
console.log(`Сумма чисел ${x} и ${y} равна: ${x + y}`);

// Пример цикла
for(let i = 0; i < 5; i++) {
    console.log(`Итерация: ${i}`);
}

// Пример массива
let arr = [1, 2, 3, 4, 5];
console.log("Элементы массива:");
for(let item of arr) {
    console.log(item);
}
